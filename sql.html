SELECT n.file_name
FROM notas n
JOIN linked_notes ln 
ON n.id = ln.note_id
JOIN notas target
ON ln.linked_note_id = target.id
WHERE target.file_name = 'CS50 - Harvard University'
AND n.path <> 'index';
